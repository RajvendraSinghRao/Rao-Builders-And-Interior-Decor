let currentIndex = 0; // Function to show the popup function showPopup(propertyId) { const popup = document.getElementById(propertyId + '-popup'); popup.style.display = 'flex'; // Load the images dynamically when the popup is opened const images = popup.querySelectorAll('.popup-image'); images.forEach(image => { image.src = image.getAttribute('data-src'); }); } // Function to close the popup function closePopup(propertyId) { const popup = document.getElementById(propertyId + '-popup'); popup.style.display = 'none'; } // Slide show logic const slides = document.querySelectorAll('.popup-content .slide'); setInterval(() => { slides[currentIndex].classList.remove('active'); currentIndex = (currentIndex + 1) % slides.length; slides[currentIndex].classList.add('active'); }, 3000);
